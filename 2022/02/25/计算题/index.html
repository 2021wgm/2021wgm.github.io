去除数组中重复的数字:

for(var i=0;i<arr.length;i++){
for(var j=i+1;j<arr.length;j++){
if(arr[i]==arr[j]){
arr.splice(j,1);
j–；此处防止出现 连续重复性数字造成漏检
}
}
}

数组排序:

arr.sort(function(a,b){
如果升序
if(a>b){
return 1（a-b）;
}else if(a<b){
return -1（a-b）;

return 0
}
如果降序:
if(a>b){
return -1（b-a);
}else if(a<b){
return 1（b-a）;
}else {
return 0;
}

判断一个数是否是质数:

质数:只能被1和它自身整除的数，1不是质数也不是和数，质数必须是大于1的自然数。
var num=prompt(“请输入一个数”);
var flag=true;
if(num<1){
alert(“请输入一个小于1的数”);
}else{
for(var i=2;i<num;i++){
//优化:i<=Math.sqrt(num) 可以对一个数开方
if(num%i==0){
flag=false; //如果num能将一个数除断,就不是一个质数
//优化：加一个break;减少时间
}
}
if(flag){
alert(num+”是一个质数”);//说明上面的num%i一直都没有除断,所以flag为false一直没执行
}else{
alert(num+”不是一个质数”);
}
